<script type="module">
    import { FaceLandmarker, FilesetResolver } from "https://cdn.jsdelivr.net/npm/@mediapipe/tasks-vision/vision_bundle.js";

    let faceLandmarker;
    let video = document.getElementById("video");
    let currentData = {};

    // 1. 初始化相機
    async function setupCamera() {
        try {
            const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "user" }, audio: false });
            video.srcObject = stream;
        } catch (e) { alert("請確保開啟相機權限並使用 HTTPS 連線。"); }
    }
    setupCamera();

    // 2. 初始化 AI 模型
    async function initAI() {
        const fileset = await FilesetResolver.forVisionTasks("https://cdn.jsdelivr.net/npm/@mediapipe/tasks-vision@latest/wasm");
        faceLandmarker = await FaceLandmarker.createFromOptions(fileset, {
            baseOptions: { modelAssetPath: `https://storage.googleapis.com/mediapipe-models/face_landmarker/face_landmarker/float16/1/face_landmarker.task`, delegate: "GPU" },
            runningMode: "IMAGE"
        });
    }
    initAI();

    // 3. 拍照與分析
    window.captureAndAnalyze = async () => {
        const canvas = document.getElementById("hidden_canvas");
        canvas.width = video.videoWidth; canvas.height = video.videoHeight;
        canvas.getContext("2d").drawImage(video, 0, 0);
        
        const result = await faceLandmarker.detect(canvas);
        if (result.faceLandmarks.length > 0) {
            runMasterDiagnosis(result.faceLandmarks[0], canvas);
            document.getElementById("page_camera").classList.remove("active");
            document.getElementById("page_result").classList.add("active");
        } else { alert("無法識別臉部，請正對相機。"); }
    };

    // 4. 深度美學診斷邏輯 (9 臉型 x 4 膚質)
    function runMasterDiagnosis(lm, canvas) {
        const getDist = (p1, p2) => Math.hypot(p1.x - p2.x, p1.y - p2.y);
        
        // 臉型偵測
        const faceH = getDist(lm[10], lm[152]), faceW = getDist(lm[234], lm[454]);
        const jawW = getDist(lm[172], lm[397]), foreheadW = getDist(lm[103], lm[332]);
        const whRatio = faceH / faceW;
        
        let shape = "鵝蛋臉";
        if (whRatio < 1.1 && jawW/faceW > 0.85) shape = "方型臉";
        else if (whRatio < 1.1) shape = "圓型臉";
        else if (whRatio > 1.3 && jawW/faceW > 0.8) shape = "矩形臉";
        else if (whRatio > 1.3) shape = "長型臉";
        else if (foreheadW > jawW * 1.2) shape = "心型臉";
        else if (jawW > foreheadW * 1.1) shape = "梨型臉";
        
        // 膚質偵測 (採樣)
        const cheekPx = canvas.getContext("2d").getImageData(lm[234].x*canvas.width, lm[234].y*canvas.height, 1, 1).data;
        let skin = "混合性";
        if (cheekPx[0] > 155) skin = "敏感性";
        else if (cheekPx[0] < 110) skin = "乾性";

        currentData = { shape, skin, name: "翊甄學員", pRatio: getDist(lm[164], lm[0])/faceH };
        
        // 更新 UI
        document.getElementById("res_face").innerText = shape;
        document.getElementById("res_skin").innerText = skin;
        document.getElementById("m_lip").innerText = currentData.pRatio > 0.16 ? "人中偏長 | 厚唇" : "比例標準";
        
        updateStyle('natural');
        updateSkincare(skin);
    }

    // 5. 彩妝風格邏輯
    window.updateStyle = (styleId) => {
        document.querySelectorAll('.style-btn').forEach(b => b.classList.remove('active'));
        const btn = [...document.querySelectorAll('.style-btn')].find(b => b.getAttribute('onclick').includes(styleId));
        if(btn) btn.classList.add('active');

        const db = {
            natural: { brow: "原生自然眉", eye: "內眼線隱形畫法", desc: "追求清透裸感，適合日常通勤。" },
            kpop: { brow: "韓系平霧眉", eye: "臥蠶重點提亮", desc: "加強臥蠶與水光感，展現減齡女神氣質。" },
            elegant: { brow: "氣質弧形眉", eye: "知性拉長眼線", desc: "強調五官對稱與高級感，適合正式場合。" }
        };
        const s = db[styleId] || db.natural;
        document.getElementById("style_tag").innerText = `風格推薦：${styleId.toUpperCase()}`;
        document.getElementById("adv_brow").innerText = s.brow;
        document.getElementById("adv_eye").innerText = s.eye;
        document.getElementById("adv_makeup_main").innerText = s.desc;
    };

    // 6. 保養與產品推薦
    function updateSkincare(skin) {
        const sop = skin === "敏感性" ? "早：溫水洗臉 > B5 噴霧 > 修復霜。晚：生理鹽水局部冷敷。" : "早：保濕潔面 > 維 C 提亮 > 防曬。晚：深層清潔 > 玻尿酸滋養。";
        document.getElementById("skincare_routine").innerText = sop;
        document.getElementById("skin_yes").innerText = skin === "敏感性" ? "積雪草, 神經醯胺" : "維他命 C, 玻尿酸";
        document.getElementById("product_list").innerHTML = `• 推薦潔顏：溫和胺基酸洗面乳<br>• 推薦精華：專屬 ${skin} 修護精華<br>• 推薦防曬：清爽型物理防曬`;
    }

    window.switchTab = (tabId) => {
        document.querySelectorAll('.tab-content').forEach(c => c.style.display = 'none');
        document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
        document.getElementById(tabId).style.display = 'block';
        event.currentTarget.classList.add('active');
    };

    // 7. 串接 GAS (請將 URL 替換為妳部署後的位址)
    window.sendReportToGas = async () => {
        const gasUrl = 'https://script.google.com/macros/s/YOUR_ACTUAL_GAS_URL/exec';
        const payload = {
            name: currentData.name,
            face_shape: currentData.shape,
            skin_type: currentData.skin,
            eyebrow: document.getElementById("adv_brow").innerText,
            skincare: document.getElementById("skincare_routine").innerText
        };
        alert("正在為您生成 PDF 醫美診斷報告...");
        // fetch(gasUrl, { method: 'POST', body: JSON.stringify(payload) });
    };
</script>
